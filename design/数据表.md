# 1. vehicles（车辆信息表）

```
CREATE TABLE IF NOT EXISTS vehicles (
    vehicle_id INTEGER PRIMARY KEY AUTOINCREMENT,
    license_plate TEXT UNIQUE NOT NULL,
    vehicle_type TEXT CHECK(vehicle_type IN ('客车','货车','特种车辆','其他')),
    owner TEXT NOT NULL,
    registration_date DATE NOT NULL
)
```
字段说明：
- vehicle_id：自增主键
- license_plate：唯一车牌号
- vehicle_type：车辆类型（枚举值）
- owner：车辆所有人
- registration_date：注册日期 

约束：
- 车牌号唯一性约束
- 车辆类型枚举检查


# 2. drivers（驾驶员信息表）
```
CREATE TABLE IF NOT EXISTS drivers (
    driver_id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    license_number TEXT UNIQUE NOT NULL,
    contact_phone TEXT,
    valid_until DATE NOT NULL,
    current_score INTEGER DEFAULT 12 CHECK(current_score >= 0 AND current_score <= 12)
)
```
字段说明：
- driver_id：自增主键
- license_number：唯一驾驶证号
- valid_until：有效期
- current_score：当前积分（默认12分）

约束：
- 驾驶证号唯一性
- 积分范围检查（0-12分）


# 3. roads（道路信息表）
```
CREATE TABLE IF NOT EXISTS roads (
    road_id INTEGER PRIMARY KEY AUTOINCREMENT,
    road_name TEXT NOT NULL,
    start_point TEXT NOT NULL,
    end_point TEXT NOT NULL,
    speed_limit INTEGER NOT NULL,
    status TEXT CHECK(status IN ('畅通','拥堵','施工','关闭'))
)
```
字段说明：
- speed_limit：限速值
- status：道路状态（枚举值）

约束：
- 状态枚举检查


# 4. violations（违章记录表）
```
CREATE TABLE IF NOT EXISTS violations (
    violation_id INTEGER PRIMARY KEY AUTOINCREMENT,
    vehicle_id INTEGER,
    driver_id INTEGER,
    violation_time DATETIME NOT NULL,
    location TEXT NOT NULL,
    violation_type TEXT NOT NULL,
    status TEXT CHECK(status IN ('未处理','已处理','申诉中')),
    FOREIGN KEY (vehicle_id) REFERENCES vehicles(vehicle_id),
    FOREIGN KEY (driver_id) REFERENCES drivers(driver_id)
)
```
关键字段：
- violation_time：违章时间
- location：违章地点
- violation_type：违章类型

关联关系：
- 外键关联车辆和驾驶员表
- 状态枚举检查


# 5. traffic_devices（交通设备表）
```
CREATE TABLE IF NOT EXISTS traffic_devices (
    device_id INTEGER PRIMARY KEY AUTOINCREMENT,
    device_type TEXT CHECK(device_type IN ('摄像头','信号灯','测速仪','电子牌')),
    location TEXT NOT NULL,
    installation_date DATE NOT NULL,
    status TEXT CHECK(status IN ('正常','故障','维修中','离线'))
)
```
设备类型：
- 包含4种交通设备类型

状态监控：
- 设备运行状态跟踪


# 6. maintenance_records（维护记录表）
```
CREATE TABLE IF NOT EXISTS maintenance_records (
    record_id INTEGER PRIMARY KEY AUTOINCREMENT,
    device_id INTEGER REFERENCES traffic_devices(device_id),
    maintenance_date DATE NOT NULL,
    operation_type TEXT CHECK(operation_type IN ('日常维护','故障维修','零件更换')),
    cost REAL
)
```
关联关系：
- 外键关联交通设备表

维护类型：
- 三种维护操作类型


# 7. operation_logs（操作日志表）
```
CREATE TABLE IF NOT EXISTS operation_logs (
    log_id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id TEXT NOT NULL,
    operation_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    action_type TEXT NOT NULL,
    detail TEXT
)
```
审计功能：
- 自动记录操作时间
- 存储用户操作详情


# 索引设计
```
CREATE INDEX IF NOT EXISTS idx_violation_time ON violations(violation_time)
CREATE INDEX IF NOT EXISTS idx_device_location ON traffic_devices(location)
CREATE INDEX IF NOT EXISTS idx_violation_location ON violations(location)
```
为高频查询字段创建索引优化性能
